\begin{lemma}
    The following are equivalent (in a locally small $\mc{C}$):
    \begin{enumerate}
        \item $f: x \to y$ is an iso in $\mc{C}$.
        \item For all $c: \obj(\mc{C})$, post-composition with $f$ defines a bijection
        \[
            f_*: \mc{C}(c, x) \to \mc{C}(c, y).
        \]
        \item For all $c: \obj(\mc{C})$, pre-composition with $f$ defines a bijection
        \[
            f_*: \mc{C}(y, c) \to \mc{C}(x, c).
        \]
    \end{enumerate}
\end{lemma}

\begin{lemma}[Compositions of mono / epi]
    \hfill
    \begin{enumerate}
        \item Composable $f$ and $g$ are mono $\Rightarrow$ $gf$ is mono.
        \item $gf$ is mono $\Rightarrow$ $f$ is mono.
        \item Composable $f$ and $g$ are epi $\Rightarrow$ $gf$ is epi.

    \end{enumerate}
