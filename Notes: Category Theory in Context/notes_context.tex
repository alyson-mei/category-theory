\begin{definition}[A representable characterization of initial or terminal objects]
    \hfill
    \begin{itemize}
        \item $c$ in $\mc{C}$ is initial iff $\mc{C}(c, -): \mc{C} \to \mc{S}et$ is isomorphic to a constant functor $\lambda -. \lbrace * \rbrace: \mc{C} \to \mc{S}et$
        \item $c \in \mc{C}$ is terminal iff the functor $\mc{C}(-, c): \mc{C}^\op \to \mc{S}et$ is iso to a functor $\lambda -. \lbrace * \rbrace: \mc{C}^\op \to \mc{S}et$
    \end{itemize}
\end{definition}

\begin{definition}[Representable functor and representation]
    \hfill
    \begin{itemize}
        \item $F: \mc{C} \to \mc{S}et$ is representable if isomorphic to $\mc{C}(c, -)$ for some $c: \obj(\mc{C})$
        \item Representation -- a choice of object with a specified isomorphism.
    \end{itemize}
\end{definition}

\begin{theorem}[Yoneda lemma]
    For any $F: \mc{C} \to \mc{S}et$, whose domain $\mc{C}$ is locally small and any object $c : \obj(\mc{C})$, there is a bijection

    that associates a natural transformation $\alpha: \mc{C}(c, -) \to F$ to the element $\alpha_c(\id_c)$. Morover, this correspondence is natural in both $c$ and $F$.
